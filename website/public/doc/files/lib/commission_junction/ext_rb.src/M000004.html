  def to_hash
    __xmlele.map do |qname, value| 
      value = if value.is_a? SOAP::Mapping::Object
                value.to_hash
              elsif value.is_a? Array
                value.map {|v| v.is_a?(SOAP::Mapping::Object) ? v.to_hash : v }
              else
                value
              end
      [qname.name, value] 
    end.to_hash
  end
